from pyscript import Element def calculate_grades(event): try: # Get values from inputs attendance = float(Element("attendance").element.value) recitation = float(Element("recitation").element.value) quiz = float(Element("quiz").element.value) requirements = float(Element("requirements").element.value) prelim_exam = float(Element("prelim_exam").element.value) # Validate input ranges if attendance < 0 or attendance > 4: Element("error").write("Grades should be between 0 and 100. Absences should be 0 or greater.") Element("error").element.style.display = "block" return if recitation < 0 or recitation > 100: Element("error").write("Grades should be between 0 and 100. Absences should be 0 or greater.") Element("error").element.style.display = "block" return if quiz < 0 or quiz > 100: Element("error").write("Grades should be between 0 and 100. Absences should be 0 or greater.") Element("error").element.style.display = "block" return if requirements < 0 or requirements > 100: Element("error").write("Grades should be between 0 and 100. Absences should be 0 or greater.") Element("error").element.style.display = "block" return if prelim_exam < 0 or prelim_exam > 100: Element("error").write("Grades should be between 0 and 100. Absences should be 0 or greater.") Element("error").element.style.display = "block" return # Calculate performance task and overall performance performance_task = recitation * 0.4 + quiz * 0.3 + requirements * 0.3 overall_performance = performance_task * 0.3 # Calculate final grade prelim_grade = round(attendance * 0.1 + overall_performance + prelim_exam * 0.6, 2) # Update the content of the result element Element("result").write(f"Your Prelim Grade is: {prelim_grade:.2f}") # Show the result section Element("result").element.style.display = "block" Element

from pyscript import Element def calculate_grades(event): try: # Get values from inputs attendance = float(Element("attendance").element.value) * 0.1 recitation = float(Element("recitation").element.value) * 0.4 quiz = float(Element("quiz").element.value) * 0.3 requirements = float(Element("requirements").element.value) * 0.3 prelim_exam = float(Element("prelim_exam").element.value) * 0.6 # Calculate performance task and overall performance performance_task = recitation + quiz + requirements overall_performance = performance_task * 0.3 # Calculate final grade prelim_grade = round(attendance + overall_performance + prelim_exam, 2) # Update the content of the result element Element("result").write(f"Your Prelim Grade is: {prelim_grade:.2f}") # Show the result section Element("result").element.style.display = "block" except ValueError: Element("result").write("Invalid input. Please enter numbers only.") Element("result").element.style.display = "block" def populate_inputs(event): # Set predefined values Element("attendance").element.value = 90 Element("recitation").element.value = 100 Element("quiz").element.value = 95 Element("requirements").element.value = 90 Element("prelim_exam").element.value = 80 # Attach event listeners to the buttons Element("calculateButton").element.onclick = calculate_grades Element("populateButton").element.onclick = populate_inputs

from pyscript import Element def calculate_grades(event): try: # Get values from inputs attendance = float(Element("attendance").element.value) * 0.1 recitation = float(Element("recitation").element.value) * 0.4 quiz = float(Element("quiz").element.value) * 0.3 requirements = float(Element("requirements").element.value) * 0.3 prelim_exam = float(Element("prelim_exam").element.value) * 0.6 # Calculate performance task and overall performance performance_task = recitation + quiz + requirements overall_performance = performance_task * 0.3 # Calculate final grade prelim_grade = round(attendance + overall_performance + prelim_exam, 2) # Calculate required grades for 75% and 90% midterm_75 = (75 - 0.2 * prelim_grade) / 0.3 final_75 = (75 - 0.2 * prelim_grade - 0.3 * midterm_75) / 0.5 midterm_90 = (90 - 0.2 * prelim_grade) / 0.3 final_90 = (90 - 0.2 * prelim_grade - 0.3 * midterm_90) / 0.5 # Update the content of the result element result_text = f"Your Prelim Grade is: {prelim_grade:.2f}\n" result_text += f"To pass with 75%, you need a Midterm grade of {midterm_75:.2f} and a Final grade of {final_75:.2f}.\n" result_text += f"To achieve 90%, you need a Midterm grade of {midterm_90:.2f} and a Final grade of {final_90:.2f}." Element("result").write(result_text) # Show the result section Element("result").element.style.display = "block" except ValueError: Element("result").write("Invalid input. Please enter numbers only.") Element("result").element.style.display = "block" # Attach event listener to the button Element("calculateButton").element.onclick = calculate_grades
